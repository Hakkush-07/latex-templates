    \vspace{\stretch{1}}

    \subpart Who actually proved the theorem?
    \vspace{\stretch{1}}
    \subpart How long did actually take to solve this problem?
    \vspace{\stretch{1}}
    \end{subparts}
    
    \end{parts}
